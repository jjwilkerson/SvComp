43  int maxSeqLength = 1;

45 
46 #ifdef DEBUG
47  int denseInSize = 20;

51  int denseInSize = 6000;
52  int numLayers = 7;
53  int sizes[] = {4000, 3000, 2000, 3000, 4000, 6000, 8000};
54 #endif
55 
56  int startEpoch = state->epoch;

58  ix_type startIx = state->sentenceIx;
59  unsigned int clockOffset = state->clock;

61 
62  Linear linear(batchSize);
63  Tanh tanh(batchSize);
64  LeakyReLU leakyRelu(batchSize);
65 
66  VecInputSource trainSource(trainCorpus, batchSize, svSize, 1, 1, startIx);
67  VecInputSource testSource(testCorpus, batchSize, svSize, 1, 1, startIx);
68 
69  VecInputLayer inputLayer("input", handle, batchSize, svSize, &linear, 0.0f);
70 
71  LossFunction* loss = new SquaredError(batchSize, 1, handle);
72 
73  SparseWeightInit ffWeightInit(config.seed, config.ffWeightCoef, 0);
74  SparseWeightInit recWeightInit(config.seed, config.recWeightCoef, 0);
75 
76  Network net(handle, batchSize, maxSeqLength, config, *loss, ffWeightInit, recWeightInit, true);
77  net.setTrainSource(&trainSource);
78 
79  net.addInput(&inputLayer);
80 
81  DenseLayer denseIn("denseIn", handle, batchSize, denseInSize, 1, &leakyRelu, 0.0f);
82  denseIn.setPrev(&inputLayer);
83  net.addHidden(&denseIn);
84  Layer* prevLayer = &denseIn;
85 
86  SplitLayer** splitLayers = new SplitLayer*[numLayers];
87  TruncateLayer** truncateLayers = new TruncateLayer*[numLayers];
88  PadLayer** padLayers = new PadLayer*[numLayers];
89  ActivationLayer2** actLayers = new ActivationLayer2*[numLayers];
90  DenseLayer2** res0Layers = new DenseLayer2*[numLayers];
91  DenseLayer2** res1Layers = new DenseLayer2*[numLayers];
92  AddLayer** addLayers = new AddLayer*[numLayers];
93 
94  int size;
95  int prevSize = denseInSize;
96  for (int l = 0; l < numLayers; l++) {
97  size = sizes[l];
98 
99  bool bigger = false, smaller = false;
100  if (size < prevSize) {
101  smaller = true;
102  } else if (size > prevSize) {
103  bigger = true;
104  }
105 
106  splitLayers[l] = new SplitLayer(name("split", l), handle, batchSize, prevSize, 1, &linear, 0.0f);
107 
108  Layer* sizingLayer = NULL;
109  if (smaller) {
110  truncateLayers[l] = new TruncateLayer(name("truncate", l), handle, batchSize, size, 1, &linear, 0.0f);
111  padLayers[l] = NULL;
112  sizingLayer = truncateLayers[l];
113  } else if (bigger) {
114  truncateLayers[l] = NULL;
115  padLayers[l] = new PadLayer(name("pad", l), handle, batchSize, size, 1, &linear, 0.0f);
116  sizingLayer = padLayers[l];
117  } else {
118  truncateLayers[l] = NULL;
119  padLayers[l] = NULL;
120  }
121 
122  if (l == 0) {
123  actLayers[l] = NULL;
124  } else {
125  actLayers[l] = new ActivationLayer2(name("act", l), handle, batchSize, prevSize, 1, &leakyRelu);
126  }
127  res0Layers[l] = new DenseLayer2(name("res0", l), handle, batchSize, size, 1, &leakyRelu, 0.0f);
128  res1Layers[l] = new DenseLayer2(name("res1", l), handle, batchSize, size, 1, &linear, 0.0f);
129  addLayers[l] = new AddLayer(name("add", l), handle, batchSize, size, 1, &linear, 0.0f);
130 
131  splitLayers[l]->setPrev(prevLayer);
132  if (sizingLayer == NULL) {
133  addLayers[l]->setPrev2(splitLayers[l]);
134  splitLayers[l]->setNext2(addLayers[l]);
135  } else {
136  addLayers[l]->setPrev2(sizingLayer);
137  sizingLayer->setPrev(splitLayers[l]);
138  sizingLayer->next = addLayers[l];
139  splitLayers[l]->setNext2(sizingLayer);
140  }
141 
142  if (l == 0) {
143  res0Layers[l]->setPrev(splitLayers[l]);
144  } else {
145  actLayers[l]->setPrev(splitLayers[l]);
146  res0Layers[l]->setPrev(actLayers[l]);
147  }
148  res1Layers[l]->setPrev(res0Layers[l]);
149  addLayers[l]->setPrev(res1Layers[l]);
150 
151  prevLayer = addLayers[l];
152  prevSize = size;
153 
154  net.addHidden(splitLayers[l]);
155  if (sizingLayer != NULL) {
156  net.addHidden(sizingLayer);
157  }
158  if (actLayers[l] != NULL) {
159  net.addHidden(actLayers[l]);
160  }
161  net.addHidden(res0Layers[l]);
162  net.addHidden(res1Layers[l]);
163  net.addHidden(addLayers[l]);
164  }
165 
166 
167  ActivationLayer actOutLayer("actOut", handle, batchSize, prevSize, 1, &linear);
168 
169  actOutLayer.setPrev(prevLayer);
170  actOutLayer.asOutputLayer();
171 
172  net.setOutput(&actOutLayer);

189 //

196  Optimizer* optimizer = new Adam(handle, net, config, startIter, false, initDeltaFilename);
197

199
